#include <iostream>
#include <string>
#include <queue>
#include <deque>
#include <tuple>
#include <algorithm>

using namespace std;

// 시작시간 : 1521
// 맞은시간 : 1530

int main(void) {
    int t; cin >> t;
    
    while (t--) {
        int n; cin >> n; // 전화번호의 수
        vector<string> str(n);
        for(int i=0; i < n; i++){
            cin >> str[i];
        }
        sort(str.begin(), str.end());
        bool ok = true;
        
        for(int i=1; i < str.size(); i++){
            int shortSize = str[i].size() > str[i-1].size() ? str[i-1].size() : str[i].size();
            bool match = true;
            for(int j=0; j < shortSize; j++){
                if(str[i][j] != str[i-1][j]) { match = false; break;}
            }
            if(match) ok = false;
        }
        
        if(ok) cout << "YES" << "\n";
        else cout << "NO" << "\n";
    }


    return 0;
}
