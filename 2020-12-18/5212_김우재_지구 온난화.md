```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
#include <string>

using namespace std;

int r, c;
vector<vector<char>> map;
vector<vector<char>> copyMap;
int dx[]  = {0,0,1, -1};
int dy[]  = {1,-1, 0, 0};

bool isSinkable(int x, int y) {
    int count = 0;
    for(int i=0; i < 4; i++){
        if(map[x+dx[i]][y+dy[i]] == '.') count++;
    }
    return count >= 3 ? true : false;
}
void doSink(int x, int y) {
    copyMap[x][y] = '.';
}

int main(){
    cin >> r >> c;
    map.resize(r+2, vector<char>(c+2, '.'));
    copyMap.resize(r+2, vector<char>(c+2, '.'));

    for(int i=1; i <= r; i++) {
        for(int j=1; j <= c; j++) {
            cin >> map[i][j];
            copyMap[i][j] = map[i][j];
        }
    }
    
    int start_x = 1;
    int end_x = r;

    int start_y = 1;
    int end_y = c;

    for(int i=1; i <= r; i++) {
        for(int j=1; j <= c; j++) {
            if(isSinkable(i, j)){
                doSink(i, j);
            }
        }
    }
    
    for(int i=1; i <= r; i++) {
        for(int j=1; j <= c; j++) {
            cout << copyMap[i][j] << " ";
        }
        cout << "\n";
    }

    for(int i=start_x; i <= end_x; i++) {
        for(int j=start_y; j <= end_y; j++) {
            cout << copyMap[i][j] << " ";
        }
        cout << "\n";
    }
    
    return 0;
}
```
